This page describes the LCM messages transmitted by the Procman sheriffs and deputies. Together, the LCM messages define the Procman communications protocol.

Messages transmitted by deputy

The deputy transmits several types of messages:

LCM Channel Message type Description
PMD_INFO2 bot_procman.info2_t Summarizes the state of all commands managed by a deputy. Transmitted at 1 Hz or when a command status changes.
PMD_PRINTF bot_procman.printf_t Published when a hosted command generates output to stdout or stderr, and contains the output produced by the command.
PMD_DISCOVER bot_procman.discovery_t Published to check for conflicting deputies with the same ID. Published when a deputy first starts up. When a deputy receives a PMD_DISCOVER message, it is to reply by transmitting its state on PMD_INFO2.

Messages transmitted by sheriff

LCM Channel Message type Description
PMD_ORDERS2 bot_procman.orders2_t Commands a deputy. Each orders message contains the desired state of all commands managed by a single deputy. To command multiple deputies, the sheriff sends one orders message to each deputy. Transmitted at 1 Hz or when a command's desired status changes.
PMD_DISCOVER bot_procman.discovery_t Published by a sheriff to discover deputies when the sheriff first starts up.

Appendix - message definitions

bot_procman.info2_t

package bot_procman;
/*
message sent by a procman deputy, primarily intended for the procman
sheriff. informs the sheriff of the status of processes running on the
host managed by the deputy.
*/
struct info2_t {
int64_t utime;
string host;
// [0, 1]
float cpu_load;
int64_t phys_mem_total_bytes;
int64_t phys_mem_free_bytes;
int64_t swap_total_bytes;
int64_t swap_free_bytes;
int32_t ncmds;
deputy_cmd2_t cmds[ncmds];
int32_t num_options;
string option_names[num_options];
string option_values[num_options];
}

bot_procman.deputy_cmd2_t

package bot_procman;
/* describes the state of a command managed by the procman sheriff/deputy
*/
struct deputy_cmd2_t {
// The command to execute.
command2_t cmd;
// If the command is running, then this is the pid if not, then this is
// zero.
int32_t pid;
// an ID for the run instance
int32_t actual_runid;
// the last exit code
int32_t exit_code;
// [0, 1]
float cpu_usage;
// total virtual memory used by the process
int64_t mem_vsize_bytes;
// total physical memory used by the process
int64_t mem_rss_bytes;
// assigned by the sheriff, identifies the command
int32_t sheriff_id;
}

bot_procman.command2_t

package bot_procman;
struct command2_t {
// Executable string
string exec_str;
string command_name;
string group;
boolean auto_respawn;
int8_t stop_signal;
float stop_time_allowed;
int32_t num_options;
string option_names[num_options];
string option_values[num_options];
}

bot_procman.printf_t

package bot_procman;
/* message sent by a procman deputy when a child process writes something to
* its stdout/stderr fd. this message contains the contents of that write
* (usually a printf)
* also sent when the deputy itself has something to say.
*/
struct printf_t {
int64_t utime;
string deputy_name;
// sheriff-assigned id of the child process. 0 if generated by the deputy
int32_t sheriff_id;
string text;
}

bot_procman.discovery_t

package bot_procman;
/*
message sent by a procman deputy or sheriff to discover other deputies and
sheriffs on the network.
*/
struct discovery_t {
int64_t utime;
string host;
int64_t nonce;
}
